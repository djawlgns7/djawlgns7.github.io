import './App.css' import Count from "./components/Count.jsx"; import {useEffect, useRef, useState} from "react"; import Buttons from "./components/Buttons.jsx"; function App() { const [count, setCount] = useState(0); const isMounted = useRef(false); // useEffect(function, deps(배열형태)) useEffect(() => { if(!isMounted.current){ isMounted.current = true; return; } console.log("mount"); }, []); useEffect(() => { console.log(`count: ${count}`); }, [count]); function modifyCount(e) { setCount(count + Number(e.target.value)); } function resetCount() { setCount(0); } return ( <> <div className={"countArea"}> <Count count={count}/> <Buttons onModify={modifyCount} onReset={resetCount} setCount={setCount} /> </div> </> ) } export default App
