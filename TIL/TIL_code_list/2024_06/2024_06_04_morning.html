이름과 이메일을 입력할 경우 입력 칸 밑에 입력한 내용을 출력해주는 코드
import {useState} from "react"; const Join2 = () => { const [input, setInput] = useState({ name: "", email: "" }); const onChangeName = (e) => { setInput({ ...input, name: e.target.value, }); } const onChangeEmail = (e) => { setInput({ ...input, email: e.target.value, }); } return ( <div> <div> <input onChange={onChangeName} placeholder={"이름써라"} value={input.name}/> <div/> {input.name} </div> <div> <input type={"email"} onChange={onChangeEmail} placeholder={"이메일써라"} value={input.email}/> <div/> {input.email} </div> </div> ); } export default Join2;
