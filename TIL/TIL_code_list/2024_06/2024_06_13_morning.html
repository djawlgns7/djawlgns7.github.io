async로 비동기 함수로 만들기

//async: 함수를 비동기 함수로 만들어주는 키워드 // 함수가 promise를 반환하도록 변환해주는 키워드 async function getProfile() { return{ name: "홍길동", id: "gildong", }; } console.log(getProfile()); async function getProfile2() { return new Promise((resolve, reject) => { setTimeout(() => { resolve({ name: "홍길동", id: "gildong", }) }, 2000); }); } getProfile2().then((result) => { console.log(result); }) // then 메서드를 안써도 됨 -> 메서드가 호출될 때 까지 기다렸다가 결과값 저장 // await -> async 안에서만 사용 가능한 키워드 async function printProfile() { const data = await getProfile2(); console.log(data); } printProfile();
